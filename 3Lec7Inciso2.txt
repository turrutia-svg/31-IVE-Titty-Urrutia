using System;

internal class Program
{
    private static void Main(string[] args)
    {
        Console.WriteLine("Triángulos por Ángulos");
        
        Console.Write("Ingrese ángulo 1: ");
        double a1 = double.Parse(Console.ReadLine());
        
        Console.Write("Ingrese ángulo 2: ");
        double a2 = double.Parse(Console.ReadLine());
        
        Console.Write("Ingrese ángulo 3: ");
        double a3 = double.Parse(Console.ReadLine());

        double suma = a1 + a2 + a3;

        
        if (suma != 180)
        {
            Console.WriteLine("La suma debe ser 180°. Esto no es un triángulo.");
        }
        else
        {
            if (a1 == a2 && a2 == a3)
            {
                Console.WriteLine(" Triángulo Equilátero.");
            }
            else if (a1 == a2 || a1 == a3 || a2 == a3)
            {
                Console.WriteLine(" Triángulo Isósceles.");
            }
            else
            {
                Console.WriteLine(" Triángulo Escaleno.");
            }
        }
    }
}
